#### -- 테이블 삭제

```
drop table pet;
```

#### -- 테이블 만들기

```
create table pet(
	name varchar(20),
    owner varchar(20),
    species varchar(20),
    gender char(1),
    birth DATE,
    death DATE
);
```

```

  from pet

#### -- Gwem과 함께 사는 아이는?

```
select name, species, owner
  from pet
 where owner = 'Gwen';
-- null 다루기 1 : 살아있는 애들은?
select  name, birth, death
  from pet
 where death is null;
    
-- null 다루기 2 : 죽어있는 애들은?
select  name, birth, death
  from pet
 where death is not null;
```

#### -- like 검색(패턴 매칭) : 이름이 b로 시작하는 아이는?

```
select name from pet where name like 'b%';
```

#### -- like 검색(패턴 매칭) : 이름이 b로 시작하는 아이들 중에 이름이 6자인 아이는?

```
select name from pet where name like 'b______';
```

#### -- 집계(통계) 함수

```

select count(death) from pet;
select count(*) from pet where death is not null;
